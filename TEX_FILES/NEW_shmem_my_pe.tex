\bAPI{SHMEM\_MY\_PE}{Returns processing element (\ac{PE}) information.}

%Synopsis C
\synC
	  int shmem_my_pe(void);
	  int _my_pe (void);
%*\synCE

%Synopsis F
\synF 
	  INTEGER SHMEM_MY_PE, ME
	  MYPE = SHMEM_MY_PE()
	  ME = MY_PE ()
%*\synFE

%DESCRIPTION

%Arguments
\desB{
	\argRow{NONE}{}{}
}
%API Description
{
	This function returns the processing element (\ac{PE}) number of the calling
  \ac{PE}.   It accepts no arguments.	The result is an integer between 0 and
  npes - 1, where npes is the total number of \ac{PE}s executing  the  current
  program.
}
%API Description Table.
{
	\desTB{ }
	{
				\cRow{}{}
	}

	%Return Value       
  \desR{Integer - Between 0 and npes - 1}

	%NOTES
	\notesB{
		For OpenSHMEM Specification 1.1 the use of \_my\_pe() has been deprecated. Although OpenSHMEM libraries are required to support the call, application developers are encouraged to use shmem\_my\_pe() instead.
	}
}
