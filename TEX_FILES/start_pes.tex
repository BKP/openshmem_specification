\bAPI{START\_PES}{Called at the beginning of an \openshmem program to initialize the execution environment.}
\synC   %Synopisis for C API 

void start_pes(int npes);
%*\synCE    %DO NOT DELETE. THIS LINE IS NOT A COMMENT

\synF   %Synopsis for FORTRAN API

CALL START_PES(npes)
 %*\synFE   %DO NOT DELETE. THIS LINE IS NOT A COMMENT  

% Arguments table. If no arguments you can use \argRow{None}{}{} 
\desB{  
       \argRow{npes}{Unused}{ Should be set to \CONST{0}.}
}
 %API description
 {   
     The \FUNC{start\_pes} routine should be used to initialize the \openshmem execution environment.  parallel
     An \openshmem application must call to \FUNC{start\_pes} before calling any other \openshmem routine.
 }
 %API Description Table. 
{
 %Return Values     
\desR{ None. }

\notesB{ If \FUNC{start\_pes}  is  not called before any other \openshmem call, unexpected behavior may occur.}
